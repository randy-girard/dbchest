# Alternative approach using nohup with proper session detachment
# Replace the systemd-run line with this if you prefer:

# Create a wrapper script that properly detaches
cat > /tmp/dbchest_wrapper.sh << 'EOF'
#!/bin/bash
# Detach from parent process completely  
setsid /tmp/dbchest_setup.sh > /var/log/dbchest-setup.log 2>&1 < /dev/null &
disown
EOF

chmod +x /tmp/dbchest_wrapper.sh
/tmp/dbchest_wrapper.sh
